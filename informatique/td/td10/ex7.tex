			\end{prooftree}
		\]
	\item
		\[
			\begin{prooftree}
				\infer 0[Ax]{(p \land r) \to (q \land r), r, p \vdash (p \land r) \to (q \land r)}

				\infer 2[$\to$e]{p \to q, p \lor r, p \vdash q}
				\infer 1[$\lor$i,g]{p \to q, p \lor r, p \vdash q \lor r}

				\infer 1[$\lor$i,d]{p \to q, p \lor r, r \vdash q \lor r}
